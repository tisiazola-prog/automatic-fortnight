def add_vehicule( type , marque , modele , immatriculation , annee , capacite , kilometrage , image_path , nb_place , cylindre ):
    
    try:
        
        conn = sqlite3.connect(get_db_path())  #Connexion à la base de données
        conn.row_factory = sqlite3.Row  #Permet de récupérer les lignes sous forme de dictionnaire
        
        query = '''
            INSERT INTO Vehicule (type, marque, model, immatriculation, annee, capacite, kilometrage, image)
            VALUES (?, ?, ?, ?, ?, ?, ?, ? )
        '''
        param =  (type , marque, modele, immatriculation, annee, capacite, kilometrage, image_path , cylindre)
        
        cursor = conn.cursor()
        cursor.execute(query , param)  #Exécution de la requête
        
        vehicule_id = cursor.lastrowid
        
        if type == "Voiture":
            query = '''
                INSERT INTO Voiture (vehicule_id , nb_places)
                VALUES (?, ?)
            '''
            param =  (vehicule_id , nb_place )
            cursor = conn.cursor()
            cursor.execute(query , param)  
            
        elif type == "Camion":
            query = '''
                INSERT INTO Camion (vehicule_id , nb_places)
                VALUES (?, ?)
            '''
            param =  (vehicule_id , nb_place )
            cursor = conn.cursor()
            cursor.execute(query , param)  
            
        elif type == "Moto":
            query = '''
                INSERT INTO Moto (vehicule_id , cylindree)
                VALUES (?, ?)
            '''
            param =  (vehicule_id , cylindre )
            cursor = conn.cursor()
            cursor.execute(query , param)  
            
            
        conn.close()  
        
        return True
